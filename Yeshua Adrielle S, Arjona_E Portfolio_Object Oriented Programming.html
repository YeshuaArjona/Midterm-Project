public class Ternary {
    public static void main(String[] args) {
        int p = 25;
        int q = 30;

        boolean out1 = (p < q) ? true : false;
        boolean out2 = (p == q) ? true : false;

Midterm Activity #3 — ATM with Transfer Feature
Java ▼

A menu-driven ATM console app with check balance, deposit, and withdrawal features including input validation and a withdrawal limit.

Java 
package me.YeshuaArjona;
import java.util.Scanner;
public class BSIT2_4_Arjona_YeshuaAdrielle_MidtermActivity3ATMwithTransferFeature {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        double balance = 9000.00;
        int choice;
        do {
            System.out.println("\n===== ATM MENU =====");
            System.out.println("1. Check Balance");
            System.out.println("2. Deposit");
            System.out.println("3. Withdraw");
            System.out.println("4. Exit");
            System.out.print("Enter your choice: ");
            choice = scanner.nextInt();
            switch (choice) {
                case 1:
                    System.out.println("\n----- Account Balance -----");
                    System.out.printf("  Current Balance: PHP %.2f%n", balance);
                    System.out.println("---------------------------");
                    break;
                case 2:
                    System.out.print("\nEnter deposit amount: PHP ");
                    double deposit = scanner.nextDouble();
                    if (deposit <= 0) {
                        System.out.println("Invalid amount. Please enter a positive value.");
                    } else {
                        balance += deposit;
                        System.out.println("\nDeposit Successful!");
                        System.out.printf("  Amount Deposited : PHP %.2f%n", deposit);
                        System.out.printf("  New Balance      : PHP %.2f%n", balance);
                    }
                    break;
                case 3:
                    System.out.print("\nEnter withdrawal amount: PHP ");
                    double withdraw = scanner.nextDouble();
                    if (withdraw <= 0) {
                        System.out.println("Invalid amount. Please enter a positive value.");
                    } else if (withdraw > 3000) {
                        System.out.println("\nWithdrawal Limit Exceeded");
                    } else if (withdraw > balance) {
                        System.out.println("\nInsufficient Balance.");
                        System.out.printf("Your current balance is PHP %.2f%n", balance);
                    } else {
                        balance -= withdraw;
                        System.out.println("\nWithdrawal Successful!");
                        System.out.printf("  Amount Withdrawn  : PHP %.2f%n", withdraw);
                        System.out.printf("  Remaining Balance : PHP %.2f%n", balance);
                    }
                    break;
                case 4:
                    System.out.println("Bye po!!!");
                    break;
                default:
                    System.out.println("Invalid choice. Please enter 1 to 4 only.");
                    break;
            }
        } while (choice != 4);
        scanner.close();
    }
}
Midterm Activity #3 — ATM

I found this activity enjoyable because it simulated a real-life system, making coding more practical and engaging. It helped me understand how programs interact with users. Midterm Activity #4 — Student Library Borrowing System with Penalty Counter
Java ▼

A student payment system with tuition fee management, discount application for scholars, and transaction tracking.

Java 
package me.YeshuaArjona;
import java.util.Scanner;
public class StudentPaymentSystemArjona {
   public static void main(String[] args) {
       Scanner scanner = new Scanner(System.in);
       String studentName;
       String studentID;
       double tuitionFee;
       int choice;
       int transaction = 0;
       boolean discountApplied = false;

       System.out.print("Enter Student Name: ");
       studentName = scanner.nextLine();
       System.out.print("Enter Student ID: ");
       studentID = scanner.nextLine();
       System.out.print("Enter Total Tuition Fee: ");
       tuitionFee = scanner.nextDouble();

       do {
           System.out.println("\n===== PAYMENT MENU =====");
           System.out.println("1. Pay Tuition");
           System.out.println("2. Check Balance");
           System.out.println("3. Apply Discount");
           System.out.println("4. Exit");
           System.out.print("Enter your choice: ");
           choice = scanner.nextInt();
           switch (choice) {
               case 1:
                   if (tuitionFee == 0) {
                       System.out.println("No remaining balance.");
                   } else {
                       System.out.print("Enter payment amount: ");
                       double payment = scanner.nextDouble();
                       if (payment > tuitionFee) {
                           System.out.println("Invalid Payment");
                       } else {
                           tuitionFee -= payment;
                           transaction++;
                           System.out.println("Payment successful.");
                           System.out.printf("Remaining Balance: %.2f%n", tuitionFee);
                       }
                   }
                   break;
               case 2:
                   System.out.printf("Remaining Balance: %.2f%n", tuitionFee);
                   transaction++;
                   break;
               case 3:
                   if (discountApplied) {
                       System.out.println("Discount already applied.");
                   } else {
                       System.out.println("Student Type:");
                       System.out.println("1. Regular Student");
                       System.out.println("2. Scholar");
                       System.out.print("Enter your choice: ");
                       int studtype = scanner.nextInt();
                       if (studtype == 2) {
                           tuitionFee -= tuitionFee * 0.20;
                           discountApplied = true;
                           System.out.println("20% discount applied.");
                           System.out.printf("New Balance: %.2f%n", tuitionFee);
                       } else {
                           System.out.println("No discount applied.");
                       }
                   }
                   break;
               case 4:
                   System.out.println("Student Name: " + studentName);
                   System.out.println("Total Transactions: " + transaction);
                   System.out.printf("Final Balance: %.2f%n", tuitionFee);
                   break;
               default:
                   System.out.println("Invalid choice.");
                   break;
           }
       } while (choice != 4);
       scanner.close();
   }
}

Midterm Activity #5 — Personal Expense Tracker
Java ▼

A modular expense tracker using static methods to calculate totals, check budget status, and display personalized summaries.

Java 
package me.YeshuaArjona;
import java.util.Scanner;
public class ExpenseTrackerArjona {

   public static void displayProgramTitle() {
       System.out.println("Personal Expense Tracker");
   }
   public static double calculateTotalExpenses(double food, double transportation, double otherExpenses) {
       return food + transportation + otherExpenses;
   }
   public static String checkBudget(double totalExpenses, double budget) {
       if (totalExpenses > budget) {
           return "You exceeded your budget by PHP "
                  + String.format("%.2f", (totalExpenses - budget));
       } else {
           return "You are within your budget.";
       }
   }
   public static void displaySummary(double totalExpenses, String budgetStatus, double budget) {
       System.out.println("\nExpense Summary");
       System.out.printf("Total Expenses : PHP %.2f%n", totalExpenses);
       System.out.printf("Budget         : PHP %.2f%n", budget);
       System.out.println("Budget Status  : " + budgetStatus);
       double remaining = budget - totalExpenses;
       if (remaining >= 0) {
           System.out.printf("Remaining      : PHP %.2f%n", remaining);
       } else {
           System.out.printf("Deficit        : PHP %.2f%n", Math.abs(remaining));
       }
   }
   public static void displayPersonalizedMessage(String name, double totalExpenses, double budget) {
       System.out.println("\nMessage for " + name);
       if (totalExpenses <= budget) {
           System.out.println("Congrats! Di ka magastos.");
       } else {
           System.out.println("Hala! Sumobra ka na.");
       }
   }
   public static void main(String[] args) {
       Scanner scanner = new Scanner(System.in);
       displayProgramTitle();
       System.out.print("\nEnter your name: ");
       String name = scanner.nextLine();
       System.out.print("Enter your budget (PHP): ");
       double budget = scanner.nextDouble();
       System.out.println("\nEnter your expenses:");
       System.out.print("  Food           : PHP ");
       double food = scanner.nextDouble();
       System.out.print("  Transportation : PHP ");
       double transportation = scanner.nextDouble();
       System.out.print("  Other Expenses : PHP ");
       double otherExpenses = scanner.nextDouble();
       double total = calculateTotalExpenses(food, transportation, otherExpenses);
       String budgetStatus = checkBudget(total, budget);
       displaySummary(total, budgetStatus, budget);
       displayPersonalizedMessage(name, total, budget);
       scanner.close();
   }
}
Midterm Activity #5 — Personal Expense Tracker
Reflection: In this activity, I learned about methods and modular programming by creating an expense tracker. I was able to relate methods to functions in C and Python, which helped me better understand how to organize code and make programs more efficient and reusable. I realized the importance of breaking programs into smaller parts. This made my code cleaner, easier to understand, and easier to maintain.
